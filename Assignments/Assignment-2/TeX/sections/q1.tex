\subsection*{Solution}

\begin{align*}
     &
    \dot x
    =
    A x
     &

    x(0)

        1 \\
        1
    \end{bmatrix}
    \\
    \implies
     &
    \dot x
    =
    M A x
     &
    M
    =
    \begin{bmatrix}
        1 & -2 \\
        0 & 1
    \end{bmatrix}
    \\
\end{align*}
